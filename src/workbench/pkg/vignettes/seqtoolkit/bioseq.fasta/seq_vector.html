Examples

 imports "bioseq.fasta" from "seqtoolkit";
 
 # get fasta sequence data
 let seqs = read.fasta("./proteins.fa");
 let sgt = seq_sgt(moltype = "prot");
 let vec = sgt |> seq_vector(seqs);
 
 # run data analysis on the generated embedding vectors
