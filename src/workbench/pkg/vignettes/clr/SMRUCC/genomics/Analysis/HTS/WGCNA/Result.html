# namespace SMRUCC.genomics.Analysis.HTS.WGCNA
export class Result {
   # 最佳beta值测试结果
   beta: BetaTest;
   # 基因显著性结果列表
   #  存储每个基因与每个表型的相关性分析结果
   geneSignificance: iterates(GeneSignificanceResult);
   # 层次聚类树
   hclust: Cluster;
   # 连接度向量
   K: Vector;
   # 模块特征基因详细结果列表
   #  包含方差解释比例等信息
   moduleEigengeneResults: iterates(ModuleEigengeneResult);
   # 模块特征基因字典（模块名→特征基因值数组）
   #  每个模块的特征基因是该模块内所有基因表达值的第一主成分
   moduleEigengenes: list;
   # 模块成员结果列表
   #  存储每个基因与每个模块特征基因的相关性
   moduleMembership: iterates(ModuleMembershipResult);
   # 模块与表型相关性结果列表
   #  存储每个模块与每个表型的相关性分析结果
   modulePhenotypeCorrelations: iterates(ModulePhenotypeCorrelation);
   # 模块字典（模块名→基因列表）
   modules: list;
   # 网络图对象
   network: NetworkGraph;
   # 所有beta候选值的测试结果
   softBeta: BetaTest[];
   # TOM矩阵
   TOM: GeneralMatrix;
}
