entry {SMRUCC.genomics.Assembly.Uniprot.XML} .NET clr documentation

entry

Description

Describes a UniProtKB entry. (因为@``P:SMRUCC.genomics.Assembly.Uniprot.XML.entry.accessions``可能会出现多个值，所以会需要使用 @``M:SMRUCC.genomics.Assembly.Uniprot.XML.entry.ShadowCopy``函数来解决实体多态的问题。 经过shadow copy之后可以使用主键@``P:SMRUCC.genomics.Assembly.Uniprot.XML.entry.accession``来创建字典)

Declare

            
# namespace SMRUCC.genomics.Assembly.Uniprot.XML
export class entry {
   dataset: string;
   created: string;
   modified: string;
   version: string;
   # 蛋白质的唯一标识符，可以用作为字典的键名
   accessions: string;
   name: string;
   protein: protein;
   features: feature[];
   gene: gene;
   proteinExistence: value;
   organism: organism;
   # 当前的这个蛋白质的蛋白序列，在Uniprot数据库之中，蛋白记录只有蛋白序列
   #  没有核酸序列，如果需要核酸序列，则会需要通过accession编号从nt库之中
   #  提取出来
   sequence: sequence;
   keywords: value[];
   comments: comment[];
   # Dictionary table can be read from @``P:SMRUCC.genomics.Assembly.Uniprot.XML.entry.xrefs`` property
   dbReferences: dbReference[];
   references: reference[];
   # indexed by @``P:SMRUCC.genomics.Assembly.Uniprot.XML.comment.type``
   CommentList: list;
   # @``P:SMRUCC.genomics.Assembly.Uniprot.XML.entry.dbReferences`` table
   xrefs: list;
}
