 `Identifier` varchar(255) DEFAULT NULL,

 `Parameter_DefaultValue` bigint(20) DEFAULT NULL,
 `Parameter_DataType` bigint(20) DEFAULT NULL,
 `Parameterizable_ParameterTypes` bigint(20) DEFAULT NULL,
